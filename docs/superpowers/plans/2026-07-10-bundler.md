# `@fixerframework/bundler` Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship `@fixerframework/bundler` — a CLI (`fixer-bundle`) plus consumable Vite/Vitest config factories that run format → lint → typecheck → test → build for lib, app, and server targets.

**Architecture:** Fail-fast pipeline of discrete steps (spawn oxfmt/oxlint/tsgo|tsc/vitest, then Vite build). Shared config factories live under export paths packages can import; no monorepo-wide forced config in v1. Typecheck prefers `@typescript/native-preview` (`tsgo`) with `tsc` fallback.

**Tech Stack:** Bun workspaces, Vite 8, Rolldown, oxfmt, oxlint, Vitest 4, TypeScript + `@typescript/native-preview` (catalog pins from root `package.json`).

**Spec:** `docs/superpowers/specs/2026-07-10-bundler-design.md`

```ts
// packages/bundler/src/resolve-bin.ts
import { existsSync } from "node:fs";
import { dirname, join } from "node:path";
import { createRequire } from "node:module";
import { fileURLToPath } from "node:url";

const require = createRequire(import.meta.url);

/**
 * Resolve a binary from a package (e.g. "oxfmt", "vitest", "vite").
 * Walks from this package and from cwd.
 */
export function resolveBin(name: string, fromDirs: string[] = []): string | null {
  const candidates: string[] = [];

  for (const dir of fromDirs) {
    candidates.push(join(dir, "node_modules", ".bin", name));
  }

  try {
    const pkgJson = require.resolve(`${name}/package.json`);
    const root = dirname(pkgJson);
    candidates.push(join(root, "bin", name));
    // common patterns
    candidates.push(join(root, "bin", `${name}.js`));
    candidates.push(join(root, "bin", `${name}.cjs`));
    candidates.push(join(root, "bin", `${name}.mjs`));
  } catch {
    // package not resolvable
  }

  // native-preview ships tsgo
  if (name === "tsgo") {
    try {
      const pkgJson = require.resolve("@typescript/native-preview/package.json");
      const root = dirname(pkgJson);
      candidates.push(join(root, "bin", "tsgo"));
      candidates.push(join(root, "lib", "tsgo"));
    } catch {
      // ignore
    }
  }

  if (name === "tsc") {
    try {
      const pkgJson = require.resolve("typescript/package.json");
      const root = dirname(pkgJson);
      candidates.push(join(root, "bin", "tsc"));
      candidates.push(join(root, "lib", "tsc.js"));
    } catch {
      // ignore
    }
  }

  const here = dirname(fileURLToPath(import.meta.url));
  candidates.push(join(here, "..", "node_modules", ".bin", name));
  candidates.push(join(process.cwd(), "node_modules", ".bin", name));

  for (const c of candidates) {
    if (existsSync(c)) return c;
  }
  return null;
}
```

```ts
// packages/bundler/src/run.ts
import { spawn } from "node:child_process";
import type { RunCommandOptions, RunCommandResult } from "./types.ts";

export async function runCommand(
  options: RunCommandOptions,
): Promise<RunCommandResult> {
  const { command, args, cwd, env } = options;

  return new Promise((resolve) => {
    const child = spawn(command, args, {
      cwd,
      env: { ...process.env, ...env },
      stdio: "inherit",
      shell: false,
    });

    child.on("error", () => {
      resolve({ code: 1 });
    });

    child.on("close", (code) => {
      resolve({ code: code ?? 1 });
    });
  });
}

// packages/bundler/src/pipeline.ts
import type { CliOptions, StepResult } from "./types.ts";

export type PipelineSteps = {
  format: () => Promise<StepResult>;
  lint: () => Promise<StepResult>;
  typecheck: () => Promise<StepResult>;
  test: () => Promise<StepResult>;
  build: () => Promise<StepResult>;
};

export type PipelineResult = {
  code: number;
  failedStep?: string;
};

const STEP_FLAGS: Array<{
  key: keyof PipelineSteps;
  enabled: (o: CliOptions) => boolean;
}> = [
  { key: "format", enabled: (o) => o.format },
  { key: "lint", enabled: (o) => o.lint },
  { key: "typecheck", enabled: (o) => o.typecheck },
  { key: "test", enabled: (o) => o.test },
  { key: "build", enabled: (o) => o.build },
];

export async function runPipeline(
  options: CliOptions,
  steps: PipelineSteps,
): Promise<PipelineResult> {
  for (const { key, enabled } of STEP_FLAGS) {
    if (!enabled(options)) continue;

    console.log(`[fixer-bundle] ${key}…`);
    const result = await steps[key]();
    if (result.code !== 0) {
      console.error(`[fixer-bundle] ${key} failed (exit ${result.code})`);
      return { code: result.code, failedStep: key };
    }
  }
  return { code: 0 };
}
```

// packages/bundler/src/cli.ts
import { resolve } from "node:path";
import type { BuildMode, CliOptions } from "./types.ts";

export type ParseSuccess = { ok: true; options: CliOptions };
export type ParseFailure = { ok: false; exitCode: number; message: string };
export type ParseResult = ParseSuccess | ParseFailure;

const MODES = new Set<BuildMode>(["lib", "app", "server"]);

export function parseArgs(argv: string[]): ParseResult {
  let mode: BuildMode | undefined;
  let cwd = process.cwd();
  let format = true;
  let lint = true;
  let typecheck = true;
  let test = true;
  let build = true;
  let formatCheck = false;
  let watch = false;
  let config: string | undefined;

  for (let i = 0; i < argv.length; i++) {
    const arg = argv[i];
    if (arg === undefined) continue;

    switch (arg) {
      case "--mode": {
        const value = argv[++i];
        if (!value || !MODES.has(value as BuildMode)) {
          return {
            ok: false,
            exitCode: 2,
            message: `Invalid or missing --mode. Expected lib|app|server, got: ${value ?? "(empty)"}`,
          };
        }
        mode = value as BuildMode;
        break;
      }
      case "--cwd": {
        const value = argv[++i];
        if (!value) {
          return { ok: false, exitCode: 2, message: "--cwd requires a path" };
        }
        cwd = resolve(value);
        break;
      }
      case "--config": {
        const value = argv[++i];
        if (!value) {
          return { ok: false, exitCode: 2, message: "--config requires a path" };
        }
        config = resolve(cwd, value);
        break;
      }
      case "--no-format":
        format = false;
        break;
      case "--no-lint":
        lint = false;
        break;
      case "--no-typecheck":
        typecheck = false;
        break;
      case "--no-test":
        test = false;
        break;
      case "--no-build":
        build = false;
        break;
      case "--format-check":
        formatCheck = true;
        break;
      case "--watch":
        watch = true;
        break;
      case "--help":
      case "-h":
        return {
          ok: false,
          exitCode: 0,
          message: `Usage: fixer-bundle --mode <lib|app|server> [options]

Options:
  --cwd <path>       Package root (default: cwd)
  --config <path>    Vite config path
  --no-format        Skip oxfmt
  --no-lint          Skip oxlint
  --no-typecheck     Skip typecheck
  --no-test          Skip vitest
  --no-build         Quality gate only
  --format-check     oxfmt check (no write)
  --watch            Watch where supported
  -h, --help         Show help
`,
        };
      default:
        return {
          ok: false,
          exitCode: 2,
          message: `Unknown argument: ${arg}`,
        };
    }
  }

  if (!mode) {
    return {
      ok: false,
      exitCode: 2,
      message: "Missing required --mode <lib|app|server>",
    };
  }

  return {
    ok: true,
    options: {
      mode,
      cwd,
      format,
      lint,
      typecheck,
      test,
      build,
      formatCheck,
      watch,
      config,
    },
  };
}
```

export async function runFormat(options: CliOptions): Promise<StepResult> {
  const bin = resolveBin("oxfmt", [options.cwd]);
  if (!bin) {
    console.error("[fixer-bundle] oxfmt not found");
    return { name: "format", code: 1 };
  }
  const args = options.formatCheck ? [".", "--check"] : [".", "--write"];
  // oxfmt CLI: prefer `oxfmt --write .` / `oxfmt --check .` — if installed version differs, adjust after smoke test
  const result = await runCommand({
    command: bin,
    args: options.formatCheck ? ["--check", "."] : ["--write", "."],
    cwd: options.cwd,
  });
  return { name: "format", code: result.code };
}

import { existsSync } from "node:fs";
import { join } from "node:path";
import type { CliOptions, StepResult } from "../types.ts";
import { resolveBin } from "../resolve-bin.ts";
import { runCommand } from "../run.ts";

function projectFlag(cwd: string): string[] {
  const build = join(cwd, "tsconfig.build.json");
  const base = join(cwd, "tsconfig.json");
  if (existsSync(build)) return ["-p", "tsconfig.build.json"];
  if (existsSync(base)) return ["-p", "tsconfig.json"];
  return [];
}

export async function runTypecheck(options: CliOptions): Promise<StepResult> {
  const project = projectFlag(options.cwd);
  const tsgo = resolveBin("tsgo", [options.cwd]);
  if (tsgo) {
    const result = await runCommand({
      command: tsgo,
      args: ["--noEmit", ...project],
      cwd: options.cwd,
    });
    // spawn success with type errors → non-zero; only fall back if binary missing (handled above)
    return { name: "typecheck", code: result.code };
  }

  const tsc = resolveBin("tsc", [options.cwd]);
  if (!tsc) {
    console.error("[fixer-bundle] neither tsgo nor tsc found");
    return { name: "typecheck", code: 1 };
  }

  // tsc.js needs node; if path ends with .js, run via node/bun
  const isJs = tsc.endsWith(".js");
  const result = await runCommand({
    command: isJs ? process.execPath : tsc,
    args: isJs ? [tsc, "--noEmit", ...project] : ["--noEmit", ...project],
    cwd: options.cwd,
  });
  return { name: "typecheck", code: result.code };
}
```

Spec: fall back on spawn failure only. If `tsgo` exists but returns type errors, do **not** fall back to tsc. If `tsgo` path exists but fails to spawn (`runCommand` error → code 1), that is still failure without silent fallback to avoid masking. Optional enhancement: detect `ENOENT` in `runCommand` and only then fall back — implement if smoke shows false negatives.

// packages/bundler/src/config/package-meta.ts
import { existsSync, readFileSync } from "node:fs";
import { join } from "node:path";

export function detectLibEntry(cwd: string): string | null {
  const candidates = [join(cwd, "index.ts"), join(cwd, "src", "index.ts")];
  for (const c of candidates) {
    if (existsSync(c)) return c;
  }
  return null;
}

export function detectServerEntry(cwd: string): string | null {
  const candidates = [
    join(cwd, "src", "server.ts"),
    join(cwd, "server.ts"),
    join(cwd, "src", "index.ts"),
    join(cwd, "index.ts"),
  ];
  for (const c of candidates) {
    if (existsSync(c)) return c;
  }
  return null;
}

export function readPackageExternals(cwd: string): Set<string> {
  const pkgPath = join(cwd, "package.json");
  const names = new Set<string>();
  if (!existsSync(pkgPath)) return names;
  const pkg = JSON.parse(readFileSync(pkgPath, "utf8")) as {
    dependencies?: Record<string, string>;
    peerDependencies?: Record<string, string>;
  };
  for (const key of Object.keys(pkg.dependencies ?? {})) names.add(key);
  for (const key of Object.keys(pkg.peerDependencies ?? {})) names.add(key);
  return names;
}

export function findLocalViteConfig(cwd: string): string | null {
  const files = [
    "vite.config.ts",
    "vite.config.mts",
    "vite.config.js",
    "vite.config.mjs",
  ];
  for (const f of files) {
    const p = join(cwd, f);
    if (existsSync(p)) return p;
  }
  return null;
}

  detectLibEntry,
  readPackageExternals,

import type { LibConfigOptions } from "./types.ts";

export function defineLibConfig(options: LibConfigOptions = {}): UserConfig {
  const {
    entry: entryOpt,
    outDir = "dist",
    cwd = process.cwd(),
    ...rest
  } = options;

  const entry = entryOpt ?? detectLibEntry(cwd);
  if (!entry) {
    throw new Error(
      `[fixer-bundle] lib mode: no entry found (index.ts or src/index.ts) in ${cwd}`,
    );
  }

  const externals = readPackageExternals(cwd);

  return defineConfig({
    build: {
      outDir,
      emptyOutDir: true,
      lib: {
        entry,
        formats: ["es"],
        fileName: "index",
      },
      rollupOptions: {
        external: (id) => {
          if (externals.has(id)) return true;
          for (const name of externals) {
            if (id === name || id.startsWith(`${name}/`)) return true;
          }
          return false;
        },
      },
      ...rest.build,
    },
    oxc: {
      jsx: {
        runtime: "automatic",
        importSource: "preact",
      },
      ...((rest as { oxc?: object }).oxc ?? {}),
    },
    ...rest,
  });
}

// packages/bundler/src/steps/build.ts
import { existsSync } from "node:fs";
import { join } from "node:path";
import { pathToFileURL } from "node:url";
import { build as viteBuild } from "vite";
import { defineAppConfig } from "../config/vite.app.ts";
import { defineLibConfig } from "../config/vite.lib.ts";
import { defineServerConfig } from "../config/vite.server.ts";
import { findLocalViteConfig } from "../config/package-meta.ts";
import type { CliOptions, StepResult } from "../types.ts";
import { resolveBin } from "../resolve-bin.ts";
import { runCommand } from "../run.ts";

async function loadConfigFromFile(configPath: string, mode: CliOptions["mode"]) {
  // Prefer Vite's config loader via programmatic API when possible.
  // For v1: dynamic import of TS config works under Bun.
  const mod = await import(pathToFileURL(configPath).href);
  return mod.default ?? mod;
}

function inlineConfigForMode(options: CliOptions) {
  switch (options.mode) {
    case "lib":
      return defineLibConfig({ cwd: options.cwd });
    case "app":
      return defineAppConfig({ cwd: options.cwd });
    case "server":
      return defineServerConfig({ cwd: options.cwd });
  }
}

export async function runBuild(options: CliOptions): Promise<StepResult> {
  try {
    const explicit = options.config;
    const local = explicit ?? findLocalViteConfig(options.cwd);

    if (local) {
      // Use vite CLI so config resolution plugins work fully
      const viteBin = resolveBin("vite", [options.cwd]);
      if (!viteBin) {
        console.error("[fixer-bundle] vite not found");
        return { name: "build", code: 1 };
      }
      const args = ["build", "--config", local];
      if (options.watch) args.push("--watch");
      const result = await runCommand({
        command: viteBin,
        args,
        cwd: options.cwd,
      });
      if (result.code !== 0) return { name: "build", code: result.code };
    } else {
      const config = inlineConfigForMode(options);
      await viteBuild({
        ...config,
        configFile: false,
        root: options.cwd,
      });
    }

    // Optional DTS: tsconfig.build.json with declaration
    const buildTsconfig = join(options.cwd, "tsconfig.build.json");
    if (existsSync(buildTsconfig)) {
      const tsc = resolveBin("tsc", [options.cwd]);
      if (tsc) {
        const isJs = tsc.endsWith(".js");
        const result = await runCommand({
          command: isJs ? process.execPath : tsc,
          args: isJs
            ? [tsc, "--emitDeclarationOnly", "-p", "tsconfig.build.json"]
            : ["--emitDeclarationOnly", "-p", "tsconfig.build.json"],
          cwd: options.cwd,
        });
        if (result.code !== 0) return { name: "build", code: result.code };
      }
    }

    return { name: "build", code: 0 };
  } catch (err) {
    console.error("[fixer-bundle] build failed:", err);
    return { name: "build", code: 1 };
  }
}

import { runFormat } from "./steps/format.ts";
import { runLint } from "./steps/lint.ts";
import { runTypecheck } from "./steps/typecheck.ts";
import { runTest } from "./steps/test.ts";

export async function main(argv: string[] = process.argv.slice(2)): Promise<number> {
  const parsed = parseArgs(argv);
  if (!parsed.ok) {
    if (parsed.message) {
      const stream = parsed.exitCode === 0 ? console.log : console.error;
      stream(parsed.message);
    }
    return parsed.exitCode;
  }

  const { options } = parsed;
  const result = await runPipeline(options, {
    format: () => runFormat(options),
    lint: () => runLint(options),
    typecheck: () => runTypecheck(options),
    test: () => runTest(options),
    build: () => runBuild(options),
  });

  return result.code === 0 ? 0 : result.code || 1;
}
